This container adapts to all screen sizes

/* Responsive CSS with Modern Features */
.container {
  max-width: 1200px;
  margin: 0 auto;
  padding: 0 1rem;
  display: grid;
  gap: 2rem;
}

@media (max-width: 768px) {
  .container {
    padding: 0 0.5rem;
    gap: 1rem;
  }
  
  .hero-title {
    font-size: clamp(2rem, 5vw, 3rem);
  }
  
  .grid {
    grid-template-columns: 1fr;
  }
}

@media (max-width: 480px) {
  .btn {
    width: 100%;
  }
  
  .section {
    padding: 2rem 0;
  }
}
